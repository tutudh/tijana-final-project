% ----------------------------------------------------------- MOTIVATION
\begin{frame}{Recap: AI-assisted labeling}
    We have a pool of unlabeled items $X_1,\dots,X_T$. Need to collect their true labels $Y_1,\dots,Y_T$.
    \medskip

    Two ways to get a label $\tilde Y_t$:
    \begin{itemize}
      \item \textbf{ML models} $\hatY_t^{(a)}$ --- sometimes \emph{wrong}, but \emph{cheap}
      \item \textbf{Expert} $Y_t$ --- \emph{correct}, but \emph{expensive}.
    \end{itemize}
    \medskip

    \textbf{Two Goals}: 
    \begin{itemize}
      \item Save cost by reducing the number of expert queries.
      \item Probably Approximately Correct (PAC) guarantee:
      \[
       \frac1T\sum_{t=1}^{T}\ell(Y_t,\tildeY_t)\;\le\;\eps\quad w.p.\ge 1-\alpha.
      \]
    \end{itemize}
\end{frame}

% --------------------------------------------------------------- SETUP
\begin{frame}{Problem setup}
  \begin{itemize}
    \item For each item $X_t$, decide how to label it: use the \textbf{expert} or one of $k$ \textbf{cheap models}.
    \item Choose action $a_t$ from an action set $\calA=\{E,1,\dots,k\}$.
    \item For item $t$ and action $a$, a labeling \textbf{loss} and a \textbf{cost}:
      \[
      h_t(a)=
      \begin{cases}0,& a=E\\[2pt] \ell(Y_t,\hatY_t^{(a)}),& a\in[k]\end{cases}
      \qquad\qquad
      c_t(a)=
      \begin{cases}1,& a=E\\[2pt] 0,& a\in[k]\end{cases}
      \]
    % \item Expert is correct ($h=0$) but costs $1$; cheap models are free
    %       ($c=0$) but may err. Losses normalized to $[0,1]$.
    \item Label the item with
          $\tildeY_t = Y_t$ if $a_t=E$, else $\hatY_t^{(a_t)}$.
  \end{itemize}
  \vfill
  \centering
  % \footnotesize
  \textcolor{accent}{Two goals: minimize \emph{cost}, control the labeling \emph{error}.}
\end{frame}

% ----------------------------------------------------------------- LP
\begin{frame}{The cost-aware labeling LP}
  Let $x_{t,a}\in[0,1]$ be the probability of taking action $a$ on item $t$.
  \[
  \begin{aligned}
    \min_{x}\;\; & \underbrace{\frac1T\sum_{t}\sum_{a} c_t(a)\,x_{t,a}}_{\text{average cost}}\\[2pt]
    \text{s.t.}\;\; & \underbrace{\frac1T\sum_{t}\sum_{a} h_t(a)\,x_{t,a}\le \eps}_{\text{average error budget}}, \qquad
    \sum_{a} x_{t,a}=1,\quad x_{t,a}\ge 0.
  \end{aligned}
  \]
  \vfill
  \begin{itemize}
    \item Models multi-sources setting and item-specific costs.
    \item If true losses $h_t(a)$ were known, this LP is the \emph{optimal} policy.
  \end{itemize}
\end{frame}

% --------------------------------------------------------- DUAL / PRICE
\begin{frame}{Dual problem and shadow price}
  One Lagrange multiplier $\lambda\ge 0$ for the error constraint. The dual:
  \[
     \max_{\lambda\ge0}\; -\lambda\eps
       +\frac1T\sum_{t=1}^{T}\min_{a\in\calA}\bigl\{c_t(a)+\lambda\,h_t(a)\bigr\}.
  \]
  \begin{block}{Simple decision rule from the shadow price}
    Given a fixed-price $\lambda$, the corresponding primal decision is given by:
    \[
       a_t(\lambda)\in\argmin_{a\in\calA}\bigl\{c_t(a)+\lambda\,h_t(a)\bigr\}.
    \]
  \end{block}

  \medskip
  \begin{itemize}
    \item {Since $h_t(a)$ is unknown, we can replace it with an estimate $\hath_t(a)$.}
    \item The PAC guarantee does not require $\hath_t(a)$ to be accurate, this affects the labeling cost.
  \end{itemize}
  

\end{frame}

% ----------------------------------------------------- THRESHOLD FORM
\begin{frame}{Price gives a threshold policy}
  Since $c_t(j)=0$ and $c_t(E)=1$  for $j\in[k]$, the decision rule is equivalent to:
  \[
     j_t^\star \in \argmin_{j\in[k]}\hath_t(j),
     \qquad s_t=\min_{j}\hath_t(j) \quad(\text{the least uncertainty score}).
  \]
  \[
     {\;s_t \ge 1/\lambda \;\Rightarrow\; \text{defer to expert}\;}
     \qquad
     s_t < 1/\lambda \;\Rightarrow\; \text{route to model $j_t^\star$}.
  \]
  \vfill
  \begin{itemize}
    \item Recovers the original \textbf{PAC-labeling} threshold rule
          (single source: $\hath_t=$ uncertainty).
    \item Multi-source setting: use least uncertain model as the routing rule.
    \item Two regimes ahead: \textbf{offline} (estimate $\hat \lambda$ on a calibration set) and
          \textbf{online} (move $\lambda_t$ over time).
  \end{itemize}
\end{frame}

\begin{frame}{Overview}
  Given a fixed-price $\lambda$, the cost-aware labeling LP gives a simple threshold policy.

\bigskip
  \centering{
  {\textcolor{accent}{Since we use proxy $\hath$, how to select $\lambda$ such that we have PAC guarantees?}}
  }
\bigskip
  \begin{itemize}
    \item \textbf{Offline} setting: fixed pool of $T$ items, estimate $\hat \lambda$ from a calibration set.
    \item \textbf{Online} setting: items arrive sequentially, we update price $\lambda_t$ onlinely.
  \end{itemize}

\end{frame}
